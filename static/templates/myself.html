function updateCounters(){ const now = new Date(); // 生日日期 6月24日 let birthday = new Date( now.getFullYear(), 5, 24, 0, 0, 0 ); if(now > birthday){ birthday = new Date( now.getFullYear()+1, 5, 24, 0, 0, 0 ); } const birthdayDiff = birthday - now; const days = Math.floor( birthdayDiff/(1000*60*60*24) ); const hours = Math.floor( (birthdayDiff/(1000*60*60))%24 ); const minutes = Math.floor( (birthdayDiff/(1000*60))%60 ); const seconds = Math.floor( (birthdayDiff/1000)%60 ); document.getElementById("countdown").innerHTML = days+" 天 "+ hours+" 小時 "+ minutes+" 分 "+ seconds+" 秒"; // 在一起日期 2023/03/19 const loveDate = new Date("2023-03-19"); const loveDiff = now - loveDate; const loveDays = Math.floor( loveDiff/(1000*60*60*24) ); document.getElementById("love-days").innerHTML = loveDays+" 天 ❤️"; } updateCounters(); setInterval(updateCounters,1000);
